import axios from 'axios';

const api = axios.create({
  baseURL: '/api',
});

export async function fetchExperiments(framework) {
  const response = await api.get(`/experiments?framework=${framework}`);
  return response.data;
}

export async function runExperiment(data) {
  const response = await api.post('/experiments', data);
  return response.data;
}

export async function fetchExperimentStatus(id) {
  const response = await api.get(`/experiments/${id}`);
  return response.data;
}

export default api;
